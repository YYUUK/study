def solution(new_id):
    answer = ''
    exc = list('~!@#$%^&*()=+[{]}:?,<>/')
    templist = list(new_id)
    removelist = []
    #1단계
    for i in range(len(templist)):
        if templist[i].isupper():
            templist[i] = templist[i].lower()
    #2단계 >> for i in templist 이 구문을 쓰면 도중에 pop, remove되면 커서가 바뀐다... 주의
    for i in templist:
        if i in exc:
            removelist.append(templist.index(i))
    print(removelist)
    while removelist:
        templist.pop(removelist.pop())
    
    print(templist)
    #3단계 >> index에서 i가 같을경우 동일한 인덱스를 두번 저장하는 문제
    iscomma = 0
    for i in templist:
        if i == '.':
            if iscomma == 1:
                removelist.append(templist.index(i))
            else:
                iscomma = 1
        else:
            iscomma = 0
    while removelist:
        templist.pop(removelist.pop())
    
    print(templist)
    return answer
